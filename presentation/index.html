Schéma de base de données


-- Création des tables principales
CREATE TABLE utilisateur (
    id INT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    pseudonyme VARCHAR(100) NOT NULL,
    motdepasse VARCHAR(255) NOT NULL,
    typecompte ENUM('redacteur', 'moderateur', 'admin') DEFAULT 'redacteur',
    estbanni BOOLEAN DEFAULT FALSE,
    datecreation DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE article (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titre VARCHAR(255) NOT NULL,
    contenu TEXT NOT NULL,
    etat ENUM('enattente', 'accepte', 'refuse') DEFAULT 'enattente',
    visibilite ENUM('public', 'prive') DEFAULT 'public',
    utilisateurid INT NOT NULL,
    datecreation DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (utilisateurid) REFERENCES utilisateur(id)
);

CREATE TABLE commentaire (
    id INT AUTO_INCREMENT PRIMARY KEY,
    contenu TEXT NOT NULL,
    etat ENUM('enattente', 'accepte', 'refuse') DEFAULT 'enattente',
    utilisateurid INT NOT NULL,
    articleid INT NOT NULL,
    datecreation DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (utilisateurid) REFERENCES utilisateur(id),
    FOREIGN KEY (articleid) REFERENCES article(id)
);

Couche Entity


class ArticleEntity {
    private int $id;
    private string $titre;
    private string $contenu;
    private string $etat; // 'enattente', 'accepte', 'refuse'
    private string $visibilite; // 'public', 'prive'
    private int $utilisateurId;
    private DateTime $dateCreation;

    public function __construct() {
        $this->dateCreation = new DateTime();
    }

    // Getters
    public function getId(): int { return $this->id; }
    public function getTitre(): string { return $this->titre; }
    public function getContenu(): string { return $this->contenu; }
    public function getEtat(): string { return $this->etat; }
    public function getVisibilite(): string { return $this->visibilite; }
    public function getUtilisateurId(): int { return $this->utilisateurId; }
    public function getDateCreation(): DateTime { return $this->dateCreation; }

    // Setters
    public function setId(int $id): void { $this->id = $id; }
    public function setTitre(string $titre): void { $this->titre = $titre; }
    public function setContenu(string $contenu): void { $this->contenu = $contenu; }
    public function setEtat(string $etat): void { $this->etat = $etat; }
    public function setVisibilite(string $visibilite): void { $this->visibilite = $visibilite; }
    public function setUtilisateurId(int $utilisateurId): void { $this->utilisateurId = $utilisateurId; }
}

Couche Service - Exemple


class ArticleService {
    private ArticleDao $articleDao;
    private UtilisateurDao $utilisateurDao;
    
    public function __construct(ArticleDao $articleDao, UtilisateurDao $utilisateurDao) {
        $this->articleDao = $articleDao;
        $this->utilisateurDao = $utilisateurDao;
    }

    public function creerArticle(string $titre, string $contenu, int $utilisateurId): ArticleEntity {
        // Validation des données
        if (empty($titre) || empty($contenu)) {
            throw new InvalidArgumentException('Le titre et le contenu sont obligatoires');
        }

        // Vérification que l'utilisateur existe et a les droits
        $utilisateur = $this->utilisateurDao->findById($utilisateurId);
        if (!$utilisateur || $utilisateur->getEstBanni()) {
            throw new UnauthorizedException('Utilisateur non autorisé');
        }

        // Création de l'entité
        $article = new ArticleEntity();
        $article->setTitre($titre);
        $article->setContenu($contenu);
        $article->setEtat('enattente'); // État par défaut
        $article->setVisibilite('public');
        $article->setUtilisateurId($utilisateurId);

        // Sauvegarde
        return $this->articleDao->create($article);
    }

    public function modererArticle(int $articleId, string $nouvelEtat, int $moderateurId): bool {
        $article = $this->articleDao->findById($articleId);
        if (!$article) {
            throw new NotFoundException('Article non trouvé');
        }

        $article->setEtat($nouvelEtat);
        $result = $this->articleDao->update($article);

        // Traçabilité
        $this->logModerationAction($moderateurId, 'article', $articleId, $nouvelEtat);

        return $result;
